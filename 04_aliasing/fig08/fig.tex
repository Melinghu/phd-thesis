\documentclass{standalone}
%
\usepackage{import}%
\input{../../preamble/standalone}%
%
\begin{document}
\begin{tikzpicture}
\pgfmathsetmacro{\xaxislength}{4.94*0.5}
\pgfmathsetmacro{\yaxislength}{3.5}

\pgfmathsetmacro{\kylength}{0.8*\yaxislength}
\pgfmathsetmacro{\kGalpha}{60}
\pgfmathsetmacro{\kGx}{\xaxislength*cos(\kGalpha)}
\pgfmathsetmacro{\kGy}{\kylength}

\pgfmathsetmacro{\deltak}{0.8}

  
\draw[-latex'] (-\xaxislength,0) -- (\xaxislength,0)
  node[pos=0.95, below]{$\sfk_{\sftangentsec}$};

\draw[-latex'] (0,0) -- (0,\yaxislength) node[at end, above]{$\sfwc$};

\draw[thick,red,smooth,samples=20,domain=0.0:\kylength]
  plot({-0.05*(exp(\x)-1)},\x);

\foreach \eta in {-2,-1,1,2}
{
  \draw[thick,dashed,red,smooth,samples=20,domain=0.0:\kylength]
    plot({\eta*\deltak-0.05*(exp(\x)-1)},\x);
}

\node[red] at (-1.0,\kylength+0.3) {\ft 
  $\sfwc \sfknorm_{\sfvirtualsource,\sftangentsec}(\sfpossec,\sfomega)$};
\node at (2*\deltak, -0.4) 
  {\ft $\frac{4\pi \sfspeedofsound}{\Delta_{\sfpossec}}$};
\node at (\deltak, -0.4) 
  {\ft $\frac{2\pi\sfspeedofsound}{\Delta_{\sfpossec}}$};
\node at (-\deltak, -0.4) 
  {\ft $\frac{-2\pi\sfspeedofsound}{\Delta_{\sfpossec}}$};
\node at (-2*\deltak, -0.4)
  {\ft $\frac{-4\pi\sfspeedofsound}{\Delta_{\sfpossec}}$};

\draw[thick,blue] (0,0) -- (\kGx,\kGy);
\node[blue] at (1.0,\kylength+0.3)
  {\ft $\sfwc \sfknorm_{\sfgreens,\sftangentsec}(\sfpos-\sfpossec)$};

\draw[dashed] (0,0) -- (\xaxislength,\yaxislength)
  node[pos=0.6, below, sloped]{\ft $\sfk_{\sftangentsec} = \sfwc$};
\draw[dashed] (0,0) -- (-\xaxislength,\yaxislength)
  node[pos=0.6, below, sloped]{\ft $\sfk_{\sftangentsec} = -\sfwc$};
%
\end{tikzpicture}
\end{document}
